}

* UserController

package com.example.controller;

import com.example.dao.UserMapper; import com.example.entity.User; import org.springframework.beans.factory.annotation.Autowired; import org.springframework.stereotype.Controller; import org.springframework.ui.Model; import org.springframework.ui.ModelMap; import org.springframework.web.bind.annotation.RequestMapping;

@Controller public class UserController {

@Autowired
private UserMapper userMapper;

@RequestMapping("/userList")
public String greetingForm(ModelMap model) {

    model.addAttribute("userList", userMapper.getUserList());
    return "userList";
}

}
